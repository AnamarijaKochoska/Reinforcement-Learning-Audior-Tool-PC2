import d4rl
import gym
import numpy as np
from typing import NamedTuple

class Transition(NamedTuple):
    observations: np.ndarray
    actions: np.ndarray
    rewards: np.ndarray
    next_observations: np.ndarray
    terminals: np.ndarray

def get_dataset(env_name: str) -> Transition:
    env = gym.make(env_name)
    raw = d4rl.qlearning_dataset(env)          # load a fixed, pre-collected benchmark dataset
    return Transition(
        observations=raw["observations"],
        actions=raw["actions"],
        rewards=raw["rewards"],
        next_observations=raw["next_observations"],
        terminals=raw["terminals"],
    )

def train(env_name: str, batch_size: int = 256):
    data = get_dataset(env_name)
    n = len(data.observations)
    for step in range(1000):
        idx = np.random.randint(0, n, size=batch_size)   # sample a minibatch from the fixed pool
        batch = Transition(*[getattr(data, f)[idx] for f in data._fields])
        # gradient update on batch; no env.step(), no live rollouts
