import numpy as np
import gymnasium as gym

class DomainRandomizationWrapper(gym.Wrapper):
    """
    Bridges the sim-to-real gap by dynamically varying physical properties
    and sensor feedback errors at each episode boundary.
    """
    def __init__(self, env, mass_range=(0.8, 1.3), friction_range=(0.5, 1.5)):
        super().__init__(env)
        self.mass_range = mass_range
        self.friction_range = friction_range

    def reset(self, **kwargs):
        # Dynamically scramble core structural properties to prevent sim over-fitting
        rand_mass = np.random.uniform(*self.mass_range)
        rand_friction = np.random.uniform(*self.friction_range)
        
        if hasattr(self.unwrapped, "model"):
            self.unwrapped.model.body_mass[1] = rand_mass
            self.unwrapped.model.geom_friction[0, 0] = rand_friction
            
        return self.env.reset(**kwargs)

    def step(self, action):
        obs, reward, terminated, truncated, info = self.env.step(action)
        # Apply artificial calibration errors to mimic degraded real-world sensors
        sensor_noise = np.random.normal(0.0, 0.02, size=obs.shape)
        corrupted_obs = obs + sensor_noise
        return corrupted_obs, reward, terminated, truncated, info
