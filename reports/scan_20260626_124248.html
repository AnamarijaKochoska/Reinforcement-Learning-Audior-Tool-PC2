def _sample_from_bounds(self, lower_bound, upper_bound, dim, attempts=5):
	"""
	Samples a point from the given bounds until all dimensions are within the bounds.
	"""
	sample = []
	for i in range(dim):
		sample.append(np.random.uniform(lower_bound, upper_bound))

	obs = np.array(sample)

	while not self._are_in_bounds(obs, lower_bound, upper_bound) and attempts > 0:
		sample = []
		for i in range(dim):
			sample.append(np.random.uniform(lower_bound, upper_bound))

		obs = np.array(sample)
		attempts -= 1

	if attempts == 0:
		obs[obs < lower_bound] = lower_bound
		obs[obs > upper_bound] = upper_bound
